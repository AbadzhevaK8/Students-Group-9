#/!bin/bash

options=(
	'1'
	"2"
	'3'
	'4'
	'5'
)
echo 'Please choose an option' // выводит сообщение - выбрать вариант (от 1 до 5)
select input in "${options[@]}"; do  
	case "$input" in
		"${options[0]}") // 1. показывает файлы находящиеся в данном каталоге
			ls
			;;
		"${options[1]}")
			date $'+Time: %T\nDate: %D' // 2. показывает текущее время и дату
			;;
			read -p 'What file do you wish to check? ' finput // 3. показывет сообщение - Какой файл хотите посмотреть? указать имя файла
		"${options[2]}")
			if [[ -d $finput ]]; then
				format='%s is a directory.\n' // при выборе наименования каталога - будет выведено сообщение.... is a directory
			elif [[ -f $finput ]]; then
				format='%s is a file.\n' // при выборе файла - будет выведено соощение - is a file
			else
				format='%s does not exist.\n' // если введено неверное имя файла или такой файл отсутствует в каталоге, будет сообщение - des not exist.
			fi
			printf "$format" "$finput"
			;;
		"${options[3]}")
			read -p 'Please enter filename: ' binput // 4. копирует выбранный в каталоге файл с добавлением расширения *.bak
			cp "$binput"{,.bak}
			;;
		"${options[4]}") // 5. закрытие скрипта
			break
			;;
		*)
			echo 'Not a valid option.' // при вводе числе вне диапозона от 1 до 5, показывает сообщение "Not a valid option."
			;;
	esac
done
